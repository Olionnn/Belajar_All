# Forcasting Data Pembelian Global Menggunakan Prophet
model = Prophet(interval_width=0.95, daily_seasonality=True)
model.fit(df)
future = model.make_future_dataframe(periods=10,freq='Y') # membuat data prediksi 10 Tahun kedepan
forecast = model.predict(future)
forecast[['ds', 'yhat', 'yhat_lower', 'yhat_upper']].tail()
