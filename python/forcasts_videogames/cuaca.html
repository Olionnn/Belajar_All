# Forcasting Data Cuaca DiAustralia Menggunakan Prophet
model = Prophet(interval_width=0.95, daily_seasonality=True)
model.fit(data)
future = model.make_future_dataframe(periods=10, freq='Y')
forecast = model.predict(future)
